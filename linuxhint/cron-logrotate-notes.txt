/home/pi/tree.txt {
    daily
    rotate 14
    maxsize 600M
    compress
    notifempty
    create
    postrotate
        rm -f ~/tree.txt
    endscript
}

pi@raspberrypi:~/repos/python-extras/linuxhint $ cat tree-cron2.sh 
#!/bin/bash

fileName="tree.txt"

cd ~/
if [[ -f "$fileName" ]]
then
        echo `tree` >> $fileName
        echo `df -h /` >> $fileName
        echo `date` >> $fileName
        echo `ls -lah ~/$fileName` >> $fileName 
else
        touch $fileName
        cp ~/tree-template.txt ~/$fileName
fi
